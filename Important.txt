CREATE TABLE users (

    name VARCHAR(100) NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT now()

message table :
query 3:
CREATE TABLE messages (
    id BIGSERIAL PRIMARY KEY,
    sender_mobile VARCHAR(15) NOT NULL REFERENCES users(mobile) ON DELETE CASCADE,
    receiver_mobile VARCHAR(15) NOT NULL REFERENCES users(mobile) ON DELETE CASCADE,
    content TEXT NOT NULL,
    is_seen BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT now()
);

-- Indexes for faster queries by receiver and sender
CREATE INDEX idx_messages_receiver ON messages(receiver_mobile);
CREATE INDEX idx_messages_sender ON messages(sender_mobile);
CREATE INDEX idx_messages_created_at ON messages(created_at);

query 4:
CREATE TABLE calls (
    id BIGSERIAL PRIMARY KEY,
    caller_mobile VARCHAR(15) NOT NULL REFERENCES users(mobile) ON DELETE CASCADE,
    receiver_mobile VARCHAR(15) NOT NULL REFERENCES users(mobile) ON DELETE CASCADE,
    offer JSONB,
    answer JSONB,
    status VARCHAR(20) DEFAULT 'ringing', -- ringing, accepted, ended
    created_at TIMESTAMP WITH TIME ZONE DEFAULT now()
);
query 5:
CREATE INDEX idx_calls_status ON calls(status);
CREATE INDEX idx_calls_caller ON calls(caller_mobile);
CREATE INDEX idx_calls_receiver ON calls(receiver_mobile);
